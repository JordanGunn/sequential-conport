---
Project environment reference. Defines workspace IDs, root paths, config files, and tool executables for reuse across workflows.
---

```yaml
name: env
context:
  workspace_id: <your_project_id>
  project:
    id: <your_project_id>
    root_path: <absolute path to project root>
    # Language-agnostic paths
    config_path: <path to main config file>
    source_path: <path to source code>
    tests_path: <path to tests>
    docs_path: <path to documentation>
    
  # Language/Framework Detection
  language: <python|javascript|go|rust|java|csharp|etc>
  framework: <poetry|npm|cargo|maven|dotnet|etc>
  
  # Generic Tool Categories (language-agnostic)
  tools:
    package_manager: <absolute path to package manager>
    test_runner: <absolute path to test runner>
    formatter: <absolute path to code formatter>
    linter: <absolute path to linter>
    type_checker: <absolute path to type checker>
    dependency_analyzer: <absolute path to dependency tool>
    build_tool: <absolute path to build tool>
    
  # Tool-specific configurations
  test:
    runner_args: ["-v", "--tb=short"]  # Default test args
    coverage_threshold: 80
    parallel: true
    
  lint:
    formatter_args: ["--check"]
    linter_args: ["--select", "E,W,F"]
    fix_safe: true
    
  # Language-specific tool profiles (examples)
  profiles:
    python:
      package_manager: poetry
      test_runner: pytest
      formatter: black
      linter: ruff
      type_checker: mypy
      dependency_analyzer: pipdeptree
      build_tool: poetry
      test_args: ["-v", "--tb=short"]
      lint_args: ["--select", "E,W,F"]
      
    javascript:
      package_manager: npm
      test_runner: jest
      formatter: prettier
      linter: eslint
      type_checker: tsc
      dependency_analyzer: npm-check
      build_tool: npm
      test_args: ["--verbose"]
      lint_args: ["--ext", ".js,.ts"]
      
    go:
      package_manager: go
      test_runner: "go test"
      formatter: gofmt
      linter: golangci-lint
      type_checker: "go vet"
      dependency_analyzer: "go mod"
      build_tool: "go build"
      test_args: ["-v", "-race"]
      lint_args: ["run"]
      
    rust:
      package_manager: cargo
      test_runner: "cargo test"
      formatter: rustfmt
      linter: clippy
      type_checker: "cargo check"
      dependency_analyzer: "cargo tree"
      build_tool: cargo
      test_args: ["--verbose"]
      lint_args: ["--", "-D", "warnings"]
      
  # Legacy compatibility (for existing workflows)
  executables: 
    poetry: <absolute path to poetry>
    git: <absolute path to git>
    # Add other tools as needed for backward compatibility
```
